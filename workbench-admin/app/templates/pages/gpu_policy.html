One-time prerequisite — JupyterHub must label pods with their group

Add this line inside build_profile() in config.yaml (inside extraConfig.dynamic-profiles), then run helm upgrade once: 

kubespawner["extra_labels"] = {"jupyterhub.io/gitlab-group": str(p.get("gitlab_group", ""))}

After this one upgrade, every notebook pod carries its group as a label. Kyverno reads that label and enforces the rules below.

  window.savePolicy = async function() {
    var msg = document.getElementById('save-msg');
    var det = document.getElementById('save-detail');
    var groups = [];
    document.querySelectorAll('#policy-rows tr:not(#empty-row)').forEach(function(tr) {
      var grp  = tr.querySelector('.grp-sel').value.trim();
      var type = tr.querySelector('.type-sel').value.trim();
      var max  = parseInt(tr.querySelector('.max-inp').value, 10);
      if (grp) groups.push({ gitlab_group: grp, gpu_type: type || 'nvidia.com/gpu', max_gpus: isNaN(max) ? 0 : max });
    });
    msg.textContent = 'Applying...'; msg.style.color = '#94a3b8';
    det.style.display = 'none';
    try {
      var d = await (await fetch('/api/gpu-policy', {
        method: 'POST', headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({ groups: groups })
      })).json();
      if (d.success) { msg.textContent = '✓ ' + d.message; msg.style.color = '#68d391'; }
      else           { msg.textContent = '✗ ' + (d.error || 'Failed'); msg.style.color = '#fc8181'; }
      if (d.detail)  { det.textContent = d.detail; det.style.display = 'block'; }
      setTimeout(loadKvStatus, 2000);
    } catch(e) { msg.textContent = '✗ Network error'; msg.style.color = '#fc8181'; }
  };

  window.loadPolicy      = loadPolicy;
  window.loadAvailGroups = window.loadAvailGroups;

  document.addEventListener('DOMContentLoaded', function() {
    if (document.getElementById('policy-rows')) { loadKvStatus(); loadPolicy(); }
  });
})();
